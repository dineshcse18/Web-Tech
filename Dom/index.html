1)WHAT IS DOM?

  • DOM is a Document Object Model.
  • It gives access to all the elements on a web page.
  • Inside the browser,the whole web page,paragraphs,tables,forms is represented in a object hierarchy.
  • Using javascript,one can create,modify and remove elements in the page dynamically.
  • The HTML DOM model is constructed as a tree of Objects:


2)DOM NODES

According to the XML DOM, everything in an XML document is a node:

  • The entire document is a document node
  • Every XML element is an element node
  • The text in the XML elements are text nodes
  • Every attribute is an attribute node
  • Comments are comment nodes

    countId=3;
    function create(){
    var para = document.createElement("p");
    var nodeId="newp"+countId;
    countId++;
    var node = document.createTextNode(document.getElementById("i1").value);
    para.appendChild(node);
    para.setAttribute("id",nodeId);
    para.setAttribute("class","highlight");
    var element = document.getElementById("div1");
    var child = document.getElementById("p1");
    element.appendChild(para,child);
    }
    function append(){
    var i=document.getElementById("i2").value;
    para=document.getElementById(i);
    para.setAttribute("class","highlight");
    var node = document.createTextNode("good day");
    para.appendChild(node);
    }
    function remove(){
    var i=document.getElementById("i3").value;
    var parent = document.getElementById("div1");
    var child = document.getElementById(i);
    parent.removeChild(child);
    }
    function insertbefore(){
    var nodeId="newp"+countId;
    countId++;
    var node = document.getElementById(document.getElementById("i4").value);
    var element = document.getElementById("div1");
    var child = document.createTextNode(" nice day");
    element.insertBefore(child,node);
    }
